# Template 'cmd.txt' for Cellranger (10X Genomics) 'atat-count'. Run 'setup_cellranger_count' to replace @VARIABLES

#e @EMAIL 
#c @CLUSTER 
#t 24
set -e

# Transcriptome reference path
INDEX=@REFVERSION

# Cellranger ATAC path
CRATAC=/tomato/dev/app/10X/cellranger-atac-2.0.0/cellranger-atac

# Input paths
FASTQS=`ls -d  MKFASTQ_*`
SAMPLE=$(echo $FASTQS | cut -d'_' -f 2)

$CRATAC count --id=$SAMPLE \
--fastqs=$FASTQS \
--reference=$INDEX \
--jobmode=local \
--localmem=95

# REMOVE temporary pipeline data
rm -rf $SAMPLE/SC_ATAC_COUNTER_CS
