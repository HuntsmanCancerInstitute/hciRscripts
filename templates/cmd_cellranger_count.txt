# Template 'cmd.txt' for Cellranger (10x Genomics) 'count'.  Run setup_cellranger.R to replace @VARIABLES

#e @EMAIL
#c @CLUSTER
#t 24
set -e

# Transcriptome reference path
INDEX=@REFVERSION

# Cellranger path 
CELLRANGER=/tomato/dev/app/10X/cellranger-6.0.1/cellranger 

# Input paths
FASTQS=`ls -d  MKFASTQ_*`
SAMPLE=$(echo $FASTQS | cut -d'_' -f 2)

$CELLRANGER count --id=$SAMPLE \
--fastqs=$FASTQS \
--transcriptome=$INDEX \
--expect-cells=@EXPECTED \
--jobmode=local

# REMOVE temporary pipeline data
rm -rf $SAMPLE/SC_RNA_COUNTER_CS/
